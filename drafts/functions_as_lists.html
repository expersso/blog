list2fun <- function(xs) { function(x) { keep(xs, ~.[[1]] == x)[[1]][[2]] } }

xs <- transpose(list(1:100, (1:100)^2)) square <- list2fun(xs) square(24)

fun2list <- function(f) { function(xs) { transpose(list(xs, map(xs, f))) } }

double <- function(x) x*2

fun2list(double)(1:10)
